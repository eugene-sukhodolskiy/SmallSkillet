This is "ss" docs (v0.1.0)

.last()
[42, 'this', 'is', 'life'].last(); // 'life'
// OR let arr = [42, 'this', 'is', 'life'];
arr.last('code');
console.log(arr); // [42, 'this', 'is', 'code']
.first()
[42, 'this', 'is', 'life'].first(); // 42
.each()
let arr = [42, 'this', 'is', 'life'].each(i, item => {
console.log(item); // show element
});
.log()
[1, 2, 3].log(); // show array to console
.different(arr)
[1, 2, 3].different([1, 5, 3]); // [2, 5]
ss('.css.selector')
ss('.css.selector'); // returned array with nodes
ss.get(url, callback, catch)
ss.get('https://give.me/json', response => {
response.log(); // response from server, log console
}, err => {
err.log(); // error
}); // this is get request to server
ss.post(url, data, callback, catch)
let data = {"name": "Adolf"}
ss.post('https://give.me/json', data, response => {
response.log(); // response from server, log console
}, err => {
err.log(); // error
}); // this is post request with data to server
.clone(obj)
let a1 = {title: "My array", meth: () => {
this.title.log();
}};
let a2 = a1.clone();
a2.log(); // {title: "My array"}
.compare(obj)
let a1 = {title: "My array", id: 3};
let a2 = a1.clone();
let a3 = {title: "Not found"};
let a4 = {title: "Not found", meth: () => {}};
a1.compare(a2); // true
a2.compare(a1); // true
a1.compare(a3); // false
a2.compare(a3); // false
a3.compare(a4); // true
.props(obj)
let a1 = {title: "My array", id: 3};
a1.props().log(); // ['title', 'id']
.addClass(className)
ss('.good.selector').last().addClass('disable');
.removeClass(className)
ss('.good.selector').first().removeClass('good');
.toggleClass(className)
ss('.good.selector').first().toggleClass('good');
.hasClass(className)
ss('.good.selector').first().hasClass('good'); // returned true
.append(element)
ss('body').append(elementORText);
.prepend(element)
ss('body').prepend(elementORText);
.on(eventName, listener)
ss('.btn').on('click', () => {
this.log(); // If you click on .btn - you will see .btn node in console
});
.find(selector)
ss('.container').first().find('h1'); // search element h1 in .container
.attr(attributeName, [value])
ss('.container').first().attr('class'); // returned value of attribute "class"
ss('.container').first().attr('class', 'no-container'); // set no-container to attribute of class
.hasAttr(attributeName)
ss('.container').first().hasAttr('class'); // returned true, if .container has attribute "class"
.show()
ss('.container').first().show(); // show element on display
.hide()
ss('.container').first().hide(); // hide element from display
.parent()
ss('.container').first().parent(); // returned parentNode
.parent()
ss('.container').first().parent(); // returned parentNode
.ss(var1, var2, ...)
"My name is {0}, I live in {1}".ss("Adolf", "Germany").log(); // show "My name is Adolf, I live in Germany"
